//Write a program that takes the size of a file (in bytes) as input from the user and converts it to the nearest whole number 
//binary prefix (B for byte, KB for kilobyte, MB for megabyte, GB for gigabyte), rounded down. Any file sizes greater than 1024 GB 
//should just be converted to GB. Remember that there are 2^10 bytes in a kilobyte (not 1000), 2^20 bytes in a megabyte, 
//and 2^30 bytes in a gigabyte.
 

num = int(input())
input_bytes = str(num)
num_temp = num
num_of_digits = 0
byte = 2**0
kilobyte = 2**10
megabyte = 2**20
gigabyte = 2**30


# counting the number of digits in the input string
for x in input_bytes:
    floor_num = num_temp // 10
    num_of_digits = num_of_digits + 1
    if(floor_num == 0):
        break
    num_temp = floor_num

# print answer in bytes if number of digits is 0-3
if((num_of_digits >= 0) and (num_of_digits <= 3)):
    result = num / byte
    print(int(result), 'B')
    
# print answer in kilobytes if number of digits is 4-6
elif((num_of_digits >= 4) and (num_of_digits <= 6)):
    result = num / kilobyte
    print(int(result), 'KB')
    
# print answer in megabytes if number of digits is 7-9
elif((num_of_digits >= 7) and (num_of_digits <= 9)):
    result = num / megabyte
    print(int(result), 'MB')
    
# print answer in gigabytes if number of digits is 10 or more
elif(num_of_digits >= 10):
    result = num / gigabyte
    print(int(result), 'GB')
